# dircetives

> 局部自定义

    //在组件中
    <div v-color:"'red"></div>
    directives:{
        //color为自定义指令名字
        color:{
            //页面加载时会触发
            mounted(el,binding){
                el.style.background = binding.value
                //els是dom元素
                //binding是自定义指令传的参数
            }，
            //页面更新时会触发
            updated(el,binding){
                el.style.background = binding.value
            }
        }
        //等价与使用mounted和updated生命周期函数
        color:(el,binding)=>{
                el.style.background = binding.value
        }
    }
